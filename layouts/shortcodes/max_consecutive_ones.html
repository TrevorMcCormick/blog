      CREATE TABLE numgroups AS
          SELECT id,
                num,
                (SELECT COUNT(*)
                    FROM nums n1
                  WHERE n1.num <> n2.num
                    AND n1.id  <= n2.id) AS num_group
            FROM nums n2;
        
      CREATE TABLE streaks AS
        SELECT num,
              MIN(id) AS startId, 
              MAX(id) AS endId, 
              COUNT(*) AS streak
          FROM numgroups
      GROUP BY num, num_group
      ORDER BY MIN(id);
      
        SELECT streak
          FROM streaks
        WHERE num = 1
      ORDER BY streak DESC
